def plot_2_var(labels,all_x,all_y,titre=None):
    # Ici, labels, all_x et all_y sont des listes de mme taille
    plt.figure(figsize=(6,6))
    for label,x,y in zip(labels,all_x,all_y):
        plt.plot(x,y,'+',label=label)
    plt.legend()
    plt.title(titre)
    plt.show()

# A complter
# Fonction de mtriques
def metrics(gt,pred,classes):
    CM = np.zeros((len(classes),len(classes)),dtype=np.int)
    top1 = 0
    top2 = 0
    for g,p in zip(gt,pred):
        if g == p[0]:
            top1+=1
            top2+=1
        elif g in p[:2]:
            top2+=1
        CM[g-1,p[0]-1]+=1
    return top1/len(gt), top2/len(gt), CM
